33 def Execute(self):
34 dt = sim.dem_solver.timestep
35
36 fx = 0
37 for w_ptr in sim.scene.wall_list:
38 fx += w_ptr.force[0]
39
40 dv = fx / self.mass * dt
41 for w_ptr in self.wall_list:
42 w_ptr.SetVelocity(w_ptr.vel[0] + dv, 0, 0)
